#!/bin/bash

source /jenkins/jt_dev/tools/python/A05_MGR_PYTHON/Compilation/etc/setenv/setenv_cmr_dev.txt

cd /jenkins/dist/FR/CMR/etc

echo 'stopping kgr server'

	echo "KGRWEB_PID_NULL"
else
	echo "KILL_KGRWEB_PID:$PIDSERV"
	kill -9 $PIDSERV
fi
