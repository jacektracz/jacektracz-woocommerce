#!/bin/bash

source /jenkins/jt_dev/tools/python/A05_MGR_PYTHON/Compilation/etc/setenv/setenv_cmr_dev.txt

cd /jenkins/dist/FR/CMR/etc

echo 'killing rvd'

PIDSERV=$(ps -ef | grep 'rvd' | grep -v 'grep'| awk '{print $2}')
if [[ -z "$PIDSERV" ]]; then
	echo "RVD_PID::NULL"
else
	echo "RVD_PID:$PIDSERV"
	kill -9 $PIDSERV
fi
echo 'killing rvd - done'
