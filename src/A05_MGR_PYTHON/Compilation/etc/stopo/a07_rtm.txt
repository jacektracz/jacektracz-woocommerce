#!/bin/bash

source /jenkins/jt_dev/tools/python/A05_MGR_PYTHON/Compilation/etc/setenv/setenv_cmr_dev.txt

echo 'stopping rtm '

cd /jenkins/dist/FR/CMR/etc/

/jenkins/dist/FR/CMR/etc/stopRTM
echo 'stopping rtm-success '

echo 'killing rtm server'

PIDSERV=$(ps -ef | grep 'rtm_ear' | grep -v 'grep'| awk '{print $2}')

if [[ -z "$PIDSERV" ]]; then
	echo "RTM_PID::NULL"
	
else
	echo "RTM_PID:$PIDSERV"
	kill -9 $PIDSERV
fi	
echo 'killing rtm - done'
