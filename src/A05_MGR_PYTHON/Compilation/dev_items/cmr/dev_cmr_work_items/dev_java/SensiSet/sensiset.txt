    @Override
    public ValueObject handle(final ValueObject valueObject) {
        try {
            switch (valueObject.getMsgId()) {
                case SensitivitySetInsertVO.MSG_ID:
                    return insertSensitivitySet((SensitivitySetInsertVO)valueObject);
                case SensitivitySetUpdateVO.MSG_ID:
                    return updateSensitivitySet((SensitivitySetUpdateVO)valueObject);
                case SensitivitySetRemoveVO.MSG_ID:
                    return removeSensitivitySet((SensitivitySetRemoveVO)valueObject);
                case SensitivitySetGetByIdentifierVO.MSG_ID:
                    return getSensitivitySetByIdentifier((SensitivitySetGetByIdentifierVO)valueObject);
                case SensitivitySetGetIdentityByIdentifierVO.MSG_ID:
                    return getSensitivitySetIdentityByIdentifier((SensitivitySetGetIdentityByIdentifierVO)valueObject);
                case SensitivitySetGetByNameVO.MSG_ID:
                    return getSensitivitySetByName((SensitivitySetGetByNameVO)valueObject);
                case SensitivitySetGetByShortNameVO.MSG_ID:
                    return getSensitivitySetByShortName((SensitivitySetGetByShortNameVO)valueObject);
                case SensitivitySetGetIdentityByNameVO.MSG_ID:
                    return getSensitivitySetIdentityByName((SensitivitySetGetIdentityByNameVO)valueObject);
                case SensitivitySetGetIdentityByShortNameVO.MSG_ID:
                    return getSensitivitySetIdentityByShortName((SensitivitySetGetIdentityByShortNameVO) valueObject);
                case SensitivitySetQueryIdentityByShortNameVO.MSG_ID:
                    return querySensitivitySetIdentityByShortName((SensitivitySetQueryIdentityByShortNameVO) valueObject);
                case SensitivitySetQueryIdentityByNameVO.MSG_ID:
                    return querySensitivitySetIdentityByName((SensitivitySetQueryIdentityByNameVO) valueObject);
                case SensitivitySetHasDataAccessVO.MSG_ID:
                    return hasSensitivitySetDataAccess((SensitivitySetHasDataAccessVO)valueObject);
                case SensitivitySetSearchDataSetTaskVO.MSG_ID:
                    return searchDataSetTask((SensitivitySetSearchDataSetTaskVO)valueObject);
                default:
                    return new GenericNACKVO(NACKException.NOTFOUND, "Unknown message id");
            }
        /*} catch (final NotFoundException e) {
            return new GenericNACKVO(NACKException.NOTFOUND, e.getMessage());*/
        } catch (final Exception e) {
            return new GenericNACKVO(NACKException.SYSTEM, e.getMessage());
        }
    }

    private GenericIdentityVectorACKVO querySensitivitySetIdentityByName(final SensitivitySetQueryIdentityByNameVO valueObject) {
        List<SensitivitySet> sensitivitySets = sensitivitySetDAO.findByNameLike(valueObject.getName(), valueObject.getMaxLength());
        return genericIdentityVectorACKTranslator.modelToVo(sensitivitySets);
    }

    private GenericIdentityVectorACKVO querySensitivitySetIdentityByShortName(final SensitivitySetQueryIdentityByShortNameVO valueObject) {
        List<SensitivitySet> sensitivitySets = sensitivitySetDAO.findByShortName(valueObject.getShortName(), valueObject.getMaxLength());
