  saveSubjects:function(requestData) {
	var self = ModDatafactory.getInstance().m_self;
    self.m_cs.mdLog("saveSubjects::method::start::");
    self.m_cs.mdLogObj(requestData ,"saveSubjects::requestData");
    
    var dd_subj_type = self.get('typeFrameData');
    
    self.m_cs.mdLogObj(dd_subj_type,"saveSubjects::dd_subj_type");
    
    var dd_subject_label = {
      type: dd_subj_type,
      streamHash: null,
      streamName: "no-m_frm_stream",
      frameFrom: 0,
      frameTo: 0
    };
    
    self.m_cs.mdLogObj(dd_subject_label,"saveSubjects::dd_subject_label");

    self.set('m_frm_subject', self.store.createRecord('subject'));
    self.get('m_frm_subject').set('rqbody', JSON.stringify(dd_subject_label));
    self.get('m_frm_subject').set('label', JSON.stringify(dd_subject_label));
    self.get('m_frm_subject').set('request', requestData);
    self.get('m_frm_subject').set('type', dd_subj_type );
    self.get('m_frm_subject').set('body',self.get('m_frm_request').get('body'));
    self.m_cs.mdLog("saveSubjects::subject_save::start::");

    var dd_subject_s =  self.get('m_frm_subject');

    self.m_cs.mdLogObj(dd_subject_s  ,"saveSubjects::dd_subject_s");

    dd_subject_s.save().then(function(subjectSaved)
    {
    	self.saveRqsctest(requestData);
    	//self.transitionToRoute();	    
    });

  },
